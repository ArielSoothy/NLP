📈 Our analysis:

Accuracy Deception

2.3Custom ROUGE Implementation & Analysis

🔧 What we implemented:

Built ROUGE-N evaluation from scratch (no external libraries).

ROUGE Implementation
Python 
def calculate_rouge_n(generated, reference, n=1):
    """Custom ROUGE-N implementation"""
    generated_ngrams = set(nltk.ngrams(generated.split(), n))
    reference_ngrams = set(nltk.ngrams(reference.split(), n))
    
    if len(reference_ngrams) == 0:
        return 0.0
    
    overlap = generated_ngrams.intersection(reference_ngrams)
    return len(overlap) / len(reference_ngrams)

def find_most_relevant_article(query_embedding, dataset, max_num_of_articles=None):
    """Find most semantically similar article using embeddings"""
    most_relevant_article = None
    max_similarity = -1
    
    for article in dataset:
        article_embedding = compute_embedding(article['text'])
        similarity = cosine_similarity(query_embedding, article_embedding)[0][0]
        
        if similarity > max_similarity:
            max_similarity = similarity
            most_relevant_article = article['text']
            
    return most_relevant_article, max_similarity
